Getting Started / Quickstart

Quickstart

Get up and running with LuxeUI in 5 minutes. This guide walks you through creating your first beautiful SwiftUI interface.

Step 1: Import & Apply Theme

Start by importing LuxeUI and applying a theme at your app's root. The theme automatically flows down to all LuxeUI components.

MyApp.swift 
import SwiftUI
import LuxeUI

@main
struct MyApp: App {
    var body: some Scene {
        WindowGroup {
            ContentView()
                .luxeTheme(.midnight)  // Apply a preset theme
        }
    }
}

Step 2: Create Your First Card

LuxeCard is a versatile container with glassmorphic styling, hover effects, and haptic feedback.

ContentView.swift 
import SwiftUI
import LuxeUI

struct ContentView: View {
    var body: some View {
        LuxeCard {
            VStack(spacing: 12) {
                Image(systemName: "star.fill")
                    .font(.system(size: 40))
                    .foregroundColor(.yellow)
                Text("Premium Feature")
                    .font(.headline)
                Text("Hover to see the effect!")
                    .font(.caption)
                    .foregroundStyle(.secondary)
            }
            .padding()
        }
        .frame(width: 280, height: 180)
    }
}

Step 5: Interactive Components

Add buttons, progress indicators, and sliders with premium styling.

Swift 
struct InteractiveExample: View {
    @State private var progress: Double = 0.65
    @State private var priceRange: [Double] = [20, 80]
    
    var body: some View {
        VStack(spacing: 32) {
            // Premium button with haptic feedback
            LuxeButton("Get Started", style: .primary) {
                print("Tapped!")
            }
            
            // Animated progress ring
            CircularProgressBar(
                progress: progress,
                showPercentage: true,
                gradient: true
            )
            .frame(width: 120, height: 120)
            
            // Multi-thumb range slider
            MultiThumbSlider(
                values: $priceRange,
                range: 0...100,
                showLabels: true
            )
            .frame(width: 300)
        }
    }
}

Complete Example

Here's everything together in a complete, beautiful interface:

ContentView.swift 
import SwiftUI
import LuxeUI

struct ContentView: View {
    @State private var progress: Double = 0.75
    
    var body: some View {
        ZStack {
            // Animated background
            MeshGradientBackground()
            
            VStack(spacing: 32) {
                // Hero card with refractive glass
                RefractiveGlassCard(configuration: .liquid) {
                    VStack(spacing: 16) {
                        CircularProgressBar(
                            progress: progress,
                            showPercentage: true
                        )
                        .frame(width: 100, height: 100)
                        
                        Text("Project Progress")
                            .font(.headline)
                    }
                    .padding()
                }
                .frame(width: 280, height: 220)
                
                // Action button
                LuxeButton("Continue", style: .primary) {
                    withAnimation {
                        progress = Double.random(in: 0.5...1.0)
                    }
                }
            }
        }
        .luxeTheme(.midnight)
    }
}
