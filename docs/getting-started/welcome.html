Quick Example

Here's how easy it is to create a stunning card with LuxeUI:

Swift 
import SwiftUI
import LuxeUI

struct ContentView: View {
    var body: some View {
        ZStack {
            MeshGradientBackground()
            
            RefractiveGlassCard {
                VStack(spacing: 16) {
                    Image(systemName: "sparkles")
                        .font(.system(size: 50))
                    Text("Liquid Glass")
                        .font(.title.bold())
                    Text("The 2026 signature effect")
                        .foregroundStyle(.secondary)
                }
            }
            .frame(width: 300, height: 220)
        }
        .luxeTheme(.midnight)
    }
}
